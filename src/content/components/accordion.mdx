## Anatomy

```tsx
import {
  Accordion,
  AccordionItem,
  AccordionTrigger,
  AccordionPanel,
} from "@/components/ui/accordion";

<Accordion>
  <AccordionItem>
    <AccordionHeader>
      <AccordionTrigger />
    </AccordionHeader>
    <AccordionPanel />
  </AccordionItem>
</Accordion>;
```
