## Add Tailwind CSS

Components are styled using Tailwind CSS. You need to install Tailwind CSS in your project.

[Follow the Tailwind CSS installation instructions to get started.](https://tailwindcss.com/docs/installation)

## Add dependencies

Add the following dependencies to your project:

<InstallationCommands
  type="package-installation"
  npmCommand="npm install @base-ui/react motion clsx tailwind-merge tailwind-variants"
  pnpmCommand="pnpm add @base-ui/react motion clsx tailwind-merge tailwind-variants"
  yarnCommand="yarn add @base-ui/react motion clsx tailwind-merge tailwind-variants"
  bunCommand="bun add @base-ui/react motion clsx tailwind-merge tailwind-variants"
/>

## Configure path aliases

Configure the path aliases in your `tsconfig.json` file:

```json filename="tsconfig.json"
{
  "compilerOptions": {
    "baseUrl": ".",
    "paths": {
      "@/*": ["./*"]
    }
  }
}
```

The `@` alias is a preference. You can use other aliases if you want.

## Add a cn helper

Add the following code to your `lib/classes.ts` file.

```ts filename="lib/classes.ts"
import { clsx, type ClassValue } from "clsx";
import { twMerge } from "tailwind-merge";

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}
```

## Create a `components.json` file

Create a `components.json` file in the root of your project.

```json filename="components.json"
{
  "$schema": "https://ui.shadcn.com/schema.json",
  "style": "new-york",
  "rsc": false,
  "tsx": true,
  "tailwind": {
    "config": "",
    "css": "src/styles/globals.css",
    "baseColor": "neutral",
    "cssVariables": true,
    "prefix": ""
  },
  "aliases": {
    "components": "@/components",
    "utils": "@/lib/utils",
    "ui": "@/components/ui",
    "lib": "@/lib",
    "hooks": "@/hooks"
  },
  "iconLibrary": "lucide",
  "registries": {
    "@pureui": "https://pure.kam-ui.com/r/{name}.json"
  }
}
```
