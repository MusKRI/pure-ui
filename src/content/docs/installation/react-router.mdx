## Create Project

<InstallationCommands
  type="package-installation"
  npmCommand="npx create-react-router@latest my-app"
  pnpmCommand="pnpm dlx create-react-router@latest my-app"
  yarnCommand="yarn create-react-router@latest my-app"
  bunCommand="bunx --bun create-react-router@latest my-app"
/>

## Run the CLI

Run the `shadcn` init command to setup your project:

<InstallationCommands
  type="package-installation"
  npmCommand="npx shadcn@latest init"
  pnpmCommand="pnpm dlx shadcn@latest init"
  yarnCommand="yarn shadcn@latest init"
  bunCommand="bunx --bun shadcn@latest init"
/>

## Add Components

You can now start adding components to your project.

<InstallationCommands type="registry-installation" registryPath="button" />

The command above will add the `Button` component to your project. You can then import it like this:

```tsx
// app/routes/home.tsx
import { Button } from "~/components/ui/button";

import type { Route } from "./+types/home";

export function meta({}: Route.MetaArgs) {
  return [
    { title: "New React Router App" },
    { name: "description", content: "Welcome to React Router!" },
  ];
}

export default function Home() {
  return (
    <div className="flex min-h-svh flex-col items-center justify-center">
      <Button>Click me</Button>
    </div>
  );
}
```
